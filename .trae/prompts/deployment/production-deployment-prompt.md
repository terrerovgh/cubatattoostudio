### 3. Rollback Procedures

#### Automated Rollback
```bash
#!/bin/bash
# rollback.sh

echo "üîÑ Initiating rollback procedure..."

# Get previous deployment
PREVIOUS_DEPLOY=$(netlify api listSiteDeploys --site-id=$NETLIFY_SITE_ID | jq -r '.[1].id')

if [ "$PREVIOUS_DEPLOY" != "null" ]; then
  echo "üì¶ Rolling back to deployment: $PREVIOUS_DEPLOY"
  netlify api restoreSiteDeploy --site-id=$NETLIFY_SITE_ID --deploy-id=$PREVIOUS_DEPLOY
  echo "‚úÖ Rollback completed successfully"
else
  echo "‚ùå No previous deployment found"
  exit 1
fi

# Verify rollback
echo "üîç Verifying rollback..."
curl -f https://cubatattoostudio.com || {
  echo "‚ùå Rollback verification failed"
  exit 1
}

echo "‚úÖ Rollback verified successfully"
```

```javascript
import { getCLS, getFID, getFCP, getLCP, getTTFB } from 'web-vitals';

function sendToAnalytics(metric) {
  // Send to Google Analytics
  gtag('event', metric.name, {
    event_category: 'Web Vitals',
    event_label: metric.id,
    value: Math.round(metric.name === 'CLS' ? metric.value * 1000 : metric.value),
    non_interaction: true,
  });
  
  // Send to custom analytics endpoint
  fetch('/api/analytics', {
    method: 'POST',
    body: JSON.stringify(metric),
    headers: {
      'Content-Type': 'application/json',
    },
  }).catch(console.error);
}

// Measure and report Core Web Vitals

getFID(sendToAnalytics);
getFCP(sendToAnalytics);
getLCP(sendToAnalytics);
getTTFB(sendToAnalytics);
```

```javascript
import * as Sentry from '@sentry/browser';
import { Integrations } from '@sentry/tracing';

Sentry.init({
```

## Security Considerations

### Security Headers Implementation
```javascript
// netlify/edge-functions/security-headers.js
export default async (request, context) => {
  const response = await context.next();
  
  // Add security headers
  response.headers.set('Strict-Transport-Security', 'max-age=31536000; includeSubDomains');
  response.headers.set('X-Frame-Options', 'DENY');
  response.headers.set('X-Content-Type-Options', 'nosniff');
  response.headers.set('X-XSS-Protection', '1; mode=block');
  response.headers.set('Referrer-Policy', 'strict-origin-when-cross-origin');
  
  return response;
};
```
